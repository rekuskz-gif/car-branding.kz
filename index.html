// ============================================ // ФАЙЛ: api/chat.js // Это сервер — он принимает сообщения от сайта, // отправляет их в Claude AI и возвращает ответ // ============================================ // ID документа Google Docs где хранится инструкция для бота const GOOGLE_DOC_ID = "1wndDcMgXu0I9H679onoXNbLzTlEiBnOzvvtQ7UYU2z4"; // Токен Телеграм бота (берётся из секретных переменных Vercel) const TG_TOKEN = process.env.TG_TOKEN; // ID чата Телеграм куда слать историю переписки const TG_CHAT = process.env.TG_CHAT; // ============================================ // ФУНКЦИЯ 1: Загружает инструкцию из Google Docs // Возвращает текст инструкции для бота // ============================================ async function loadPrompt() { try { // Специальная ссылка чтобы получить документ как чистый текст const url = `https://docs.google.com/document/d/${GOOGLE_DOC_ID}/export?format=txt`; const response = await fetch(url); // Если Google Docs не ответил — используем запасной текст if (!response.ok) { return "Ты Катя, консультант car-branding.kz"; } // Убираем лишние пробелы let text = await response.text(); text = text.trim(); // Если документ пустой — используем запасной текст return text || "Ты Катя, консультант car-branding.kz"; } catch (e) { // Если совсем что-то сломалось — используем запасной текст return "Ты Катя, консультант car-branding.kz"; } } // ============================================ // ФУНКЦИЯ 2: Отправляет историю чата в Телеграм // Вы будете видеть переписку клиентов у себя в боте // ============================================ async function sendToTelegram(messages) { try { let text = "📋 История чата:\n\n"; for (let msg of messages) { if (msg.role === "user") { text += `👤 Клиент: ${msg.content}\n\n`; } else { text += `🤖 Амина: ${msg.content}\n\n`; } } await fetch(`https://api.telegram.org/bot${TG_TOKEN}/sendMessage`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ chat_id: TG_CHAT, text: text }) }); } catch (e) { // Если Телеграм не ответил — просто пишем в лог, не ломаем чат console.error("Telegram error:", e); } } // ============================================ // ГЛАВНАЯ ФУНКЦИЯ: Обрабатывает каждое сообщение // Сюда приходят запросы с сайта // ============================================ module.exports = async (req, res) => { // Разрешаем запросы с любого сайта res.setHeader("Access-Control-Allow-Origin", "*"); res.setHeader("Access-Control-Allow-Methods", "POST, OPTIONS"); res.setHeader("Access-Control-Allow-Headers", "Content-Type"); // Браузер проверяет можно ли слать запросы — говорим да if (req.method === "OPTIONS") return res.status(200).end(); // Принимаем только POST запросы if (req.method !== "POST") return res.status(405).json({ error: "Only POST", version: "2.0", model: "claude-haiku-4-5-20251001" }); try { // Берём API ключ Claude из секретных переменных Vercel const apiKey = process.env.ANTHROPIC_API_KEY; if (!apiKey) return res.status(500).json({ error: "API Key не настроен" }); // Берём историю сообщений из запроса const { messages } = req.body; if (!messages || !Array.isArray(messages)) { return res.status(400).json({ error: "Нет сообщений" }); } // Загружаем инструкцию из Google Docs const systemPrompt = await loadPrompt(); // Отправляем запрос в Claude AI const response = await fetch("https://api.anthropic.com/v1/messages", { method: "POST", headers: { "x-api-key": apiKey, "anthropic-version": "2023-06-01", "content-type": "application/json" }, body: JSON.stringify({ model: "claude-haiku-4-5-20251001", // название модели max_tokens: 300, // максимум 300 слов в ответе system: systemPrompt, // инструкция для бота messages: messages.map(msg => ({ role: msg.role, content: msg.content })) }) }); const data = await response.json(); // Если Claude вернул ошибку if (!response.ok) { console.error("Claude API Error:", data); return res.status(response.status).json({ error: data.error?.message || "Ошибка Claude API", choices: [{ message: { content: "Извините, произошла ошибка. Попробуйте позже." } }] }); } // Достаём текст ответа const botMessage = data.content?.[0]?.text || "Извините, не могу ответить сейчас."; // Отправляем переписку в Телеграм await sendToTelegram([...messages, { role: "assistant", content: botMessage }]); // Отправляем ответ на сайт res.status(200).json({ choices: [{ message: { content: botMessage } }] }); } catch (error) { console.error("Server Error:", error); res.status(500).json({ error: "Ошибка сервера", choices: [{ message: { content: "Извините, произошла ошибка. Попробуйте позже." } }] }); } };
